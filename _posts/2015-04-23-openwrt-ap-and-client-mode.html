[code lang=text]
config wifi-device 'ra0'
option type 'rt2860v2'
option name 'rt2860v2'
option mode '9'
option txpower '100'
option ht '40'
option country 'US'
option disabled '0'
option channel '3' # same channel with AP server
[/code]

== AP Mode ==

[code lang=text]
config wifi-iface
option device 'ra0'
option network 'lan'
option mode 'ap'
option ssid 'mydreamy-br'
option encryption 'psk2+aes'
option key ''
[/code]

== Client Mode == /etc/config/dhcp - lan

[code lang=text]
config wifi-iface
option network 'wwan'
option ssid 'mydreamy'
option wds '1'
option encryption 'psk2+aes'
option device 'ra0'
option mode 'sta'
option key ''
[/code]

== Bridge Mode == /etc/config/dhcp - lan : option ignore 1

[code lang=text]
config wifi-iface
option network 'lan'
option ssid 'mydreamy'
option wds '1'
option encryption 'psk2+aes'
option device 'ra0'
option mode 'sta'
option key ''
[/code]
