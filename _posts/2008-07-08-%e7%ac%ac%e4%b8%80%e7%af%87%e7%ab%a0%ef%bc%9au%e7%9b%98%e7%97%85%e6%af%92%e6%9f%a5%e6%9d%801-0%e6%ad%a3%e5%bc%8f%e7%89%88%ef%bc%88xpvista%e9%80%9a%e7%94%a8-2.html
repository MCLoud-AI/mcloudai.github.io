先来分享一个前阵用批处理写的U盘杀毒的，原理很简单，但很实用，应该每次手杀太麻烦，所以写了他，希望跟大家分享

 @echo off
echo 进行U盘病毒免疫
reg add HKEY_CURRENT_USERSoftwareMicrosoftWindowsCurrentVersionPoliciesExplorer /v NoDriveTypeAutoRun /t REG_DWORD /d 0x91 /f
echo --------------------------------------------------------------------
if errorlevel 1 echo 免疫失败
if errorlevel 0 echo 免疫成功
echo --------------------------------------------------------------------
echo 请你插入你的U盘，并按任意键继续
pause
echo --------------------------------------------------------------------
vol c: && c:
vol d: && d:
vol e: && e:
vol f: && f:
vol g: && g:
vol h: && h:
vol i: && i:
vol j: && j:
vol k: && k:
vol l: && l:
vol m: && m:
vol n: && n:
echo --------------------------------------------------------------------
echo 正在进行杀毒........
echo --------------------------------------------------------------------
if exist autorun.inf del autorun.inf
if exist autorun.inf del /a:h /f autorun.inf
if exist autorun.inf del /a:s /f autorun.inf
if exist *.inf del /a:h /f *.inf
if exist *.inf del /a:s /f *.inf
if exist *.exe del /a:h /f *.exe
if exist *.exe del /a:s /f *.exe
if exist *.bat del /a:h /f *.bat
if exist *.bat del /a:s /f *.bat
if exist *.vbs del /a:h /f *.vbs
if exist *.vbs del /a:s /f *.vbs
if exist *.pif del /a:h /f *.pif
if exist *.pif del /a:s /f *.pif
echo --------------------------------------------------------------------

echo    by: yangyilong

echo --------------------------------------------------------------------

echo 你的U盘已经安全
echo 按任意键后重新插拔U盘
pause
