安裝bandwidthd
# cd /usr/ports/net-mgmt/bandwidthd
# make install clean
# cd /usr/local/bandwidthd/etc
# cp bandwidthd.conf-dist bandwidthd.conf
# ee bandwidthd.conf 修改如下
subnet 10.0.0.0 255.0.0. //預設值
subnet 192.168.1.0/24 //修改要統計的網段
dev "eth0″ //預設值
dev "re0″ //修改網卡
存檔
# cd /usr/local/etc/rc.d
# cp bandwidthd.sh.sample bandwidthd.sh
# /usr/local/etc/rc.d/bandwidthd.sh start
# cd /usr/local/www/apache22/data
# ln -s /usr/local/bandwidthd/htdocs bandwidthd

網址輸入http://IP/bandwidthd

 

安裝NTOP
# cd /usr/ports/net/ntop/
# make install clean
# rehash
# ntop -A
輸入admin密碼兩次
# ee /etc/rc.conf 加入
ntop_enable="YES"

ntop_flags="-d --use-syslog=daemon -4"

存檔
# /usr/local/etc/rc.d/ntop start
網址輸入http://ip:3000/
