  1. Install GCC 4.9
    1. brew install homebrew/versions/gcc49
  2. GMP
    1. ./configure --prefix=/usr/local/gmp CC=gcc-4.9
    2. make
    3. make check
    4. sudo make install
  3. NTL
    1. cd src
    2. ./configure PREFIX=/usr/local/ntl NTL_GMP_LIP=on GMP_PREFIX=/usr/local/gmp CXX=g++-4.9
    3. make
    4. sudo make install
  4. HElib
