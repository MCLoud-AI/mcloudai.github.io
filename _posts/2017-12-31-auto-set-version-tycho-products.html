<pluginManagement>

<plugins>

<plugin>

<groupId>org.eclipse.tycho</groupId>

<artifactId>tycho-versions-plugin</artifactId>

<version>${tycho-version}</version>

</plugin>

</pluginManagement>

</plugins>

Then,

mvn tycho-versions:set-version -DnewVersion=1.0.0-SNAPSHOT
